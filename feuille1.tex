\begin{exo}[Factoriser les expressions suivantes]
\begin{enumerate}
\item $6x^2 + 9x$
  \ifcorrige\noindent \color{magenta}Correction :
  $3x(2x+3)$
  \color{black}\fi
\item $12x^2 y + 8xy^2$
  \ifcorrige\noindent \color{magenta}Correction :
  $4xy(3x+2y)$
  \color{black}\fi
\item $15x - 20xy^2$
  \ifcorrige\noindent \color{magenta}Correction :
  $5x(3-4y^2)$
  \color{black}\fi
\item $xy^2 - x^2y$
  \ifcorrige\noindent \color{magenta}Correction :
  $xy(y-x)$
  \color{black}\fi
\item $x^2y - xy^3$
  \ifcorrige\noindent \color{magenta}Correction :
  $xy(x-y^2)$
  \color{black}\fi
\item $5x^2y - 25xy + 3y^2x$
  \ifcorrige\noindent \color{magenta}Correction :
  $xy(5x - 25 + 3y)$
  \color{black}\fi
\end{enumerate} 
\end{exo}
